CREATE TABLE webmeta(
	id INT AUTO_INCREMENT PRIMARY KEY,
	contents LONGTEXT,
	datetime DATETIME,
	title VARCHAR(255),
	url VARCHAR(255)
);



CREATE TABLE videometa(
	id INT AUTO_INCREMENT PRIMARY KEY,
	author VARCHAR(255),
	datetime DATETIME,
	play_time INT,
	thumbnail LONGTEXT,
	title VARCHAR(255),
	url LONGTEXT
);


CREATE TABLE reservation(
	id INT AUTO_INCREMENT PRIMARY KEY,
	keyword VARCHAR(255),
	type VARCHAR(50),
	sort VARCHAR(50),
	cron VARCHAR(100),
	status VARCHAR(100),
	createDate DATETIME,
	use_yn VARCHAR(50)
);
